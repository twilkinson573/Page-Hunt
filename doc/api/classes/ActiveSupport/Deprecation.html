def initialize(deprecation_horizon = '4.2', gem_name = 'Rails')
  self.gem_name = gem_name
  self.deprecation_horizon = deprecation_horizon
  # By default, warnings are not silenced and debugging is off.
  self.silenced = false
  self.debug = false
end

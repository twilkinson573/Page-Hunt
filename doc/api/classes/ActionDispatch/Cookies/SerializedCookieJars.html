def deserialize(name, value)
  if value
    if needs_migration?(value)
      Marshal.load(value).tap do |v|
        self[name] = { value: v }
      end
    else
      serializer.load(value)
    end
  end
end

def serializer
  serializer = @options[:serializer] || :marshal
  case serializer
  when :marshal
    Marshal
  when :json, :hybrid
    JsonSerializer
  else
    serializer
  end
end
